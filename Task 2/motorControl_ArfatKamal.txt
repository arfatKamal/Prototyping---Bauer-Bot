// Motor A connections
int enA = 10;
int in1 = 9;
int in2 = 8;
// Motor B connections
int enB = 5;
int in3 = 7;
int in4 = 6;


void setup() {
pinMode(9,OUTPUT);   //left motor forward
pinMode(8,OUTPUT);   //left motor reverse
pinMode(7,OUTPUT);   //right motor forward
pinMode(6,OUTPUT);   //right motor reverse
pinMode(enA, OUTPUT);
pinMode(enB, OUTPUT);

Serial.begin(9600);
}


//Motor Commands
forward {
 digtatWrite(in1, HIGH);
 digtatWrite(in3, HIGH);
 digitalWrite(in2, LOW);
 digitalWrite(in4, LOw);
}

backward {
 digtatWrite(in1, LOW);
 digtatWrite(in3, LOW);
 digitalWrite(in2, HIGH);
 digitalWrite(in4, HIGH);
}

forward_right {
 digtatWrite(in1, HIGH);
 digtatWrite(in3, LOW);
 digitalWrite(in2, LOW);
 digitalWrite(in4, HIGH);
}

forward_left {
 digtatWrite(in1, LOW);
 digtatWrite(in3, HIGH);
 digitalWrite(in2, HIGH);
 digitalWrite(in4, LOW);
}

stop {
 digtatWrite(in1, LOW);
 digtatWrite(in3, LOW);
 digitalWrite(in2, LOW);
 digitalWrite(in4, LOW);
}
